function hostnameForIP() {
        if [ -z "$1" ]; then
                return
        fi

        port="$(echo "$1" | awk -F: '{print $NF}')"
        ip="${1//:$port/}"
        # strip IPv6 brackets
        ip="${ip//[][]/}"
        hname="$(dig +short -x "$ip" xargs)"
        if [ -n "$hname" ]; then
                echo "${hname//\.$/}"
        else
                echo "$ip"
        fi
}
