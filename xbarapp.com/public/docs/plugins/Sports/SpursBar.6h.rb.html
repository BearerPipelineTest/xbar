require 'open-uri'
require 'nokogiri'
require 'active_support/time'

class SpursBar
  def initialize
    @font_face = "font=Spurs"
    @font_lg = "size=20 #{@font_face}"
    @font_md = "size=18 #{@font_face}"
    @font_rg = "size=16 #{@font_face}"
    @font_color = ENV['BitBarDarkMode'] ? "white" : "#465882"
    @is_usa = %w('PDT PST MDT MST CDT CST EDT EST').include? Time.now.zone
    @team = Nokogiri::HTML(open("http://www.bbc.com/sport/football/teams/tottenham-hotspur"))
  end

  def output
    puts header
    puts '---'
    puts team_name
    puts today_fixtures
    puts upcoming_fixtures
    puts previous_fixtures
    puts headlines
  end

  private

  def header
    icon_image = matchday.empty? ? icon.to_s : icon_matchday.to_s
    "| templateImage=\"#{icon_image}\""
  end

  def team_name
    name = @team.css('.headline-wrapper h1').first.text.upcase
    puts "#{name} | #{@font_lg} color=#{@font_color}"
    puts '---'
  end

  def today_fixtures
    item = @team.css('.accordion-container.live-today')
    unless item.empty?
      puts "TODAY'S FIXTURE âš½ï¸Ž COYS | #{@font_lg} color=#{@font_color}" if item

      date = Date.today
      fixture = item.css("ul li").first
      link = fixture.css("a").first['href']
      time = fixture.css(".status.upcoming abbr")
      kickoff = ""
      unless time.empty?
        datetime = "#{date}T#{time.first.text}" unless time.nil?
        kickoff = " â€£ #{fixture_time(datetime).upcase}" unless time.nil?
      end

      puts "#{fixture.text.strip.gsub(/(\s*\d{2}:\d{2})/, '').upcase}#{kickoff}  | #{@font_md} color=#{@font_color}"
      puts "--MATCH INFO âš½ï¸Ž | #{@font_md} color=#{@font_color}"
      puts "--#{fixture_date(date)}"
      puts "--BBC Sport Preview ðŸ“° | href=http://www.bbc.com#{link}"
      puts "---"
    end
  end

  def upcoming_fixtures
    fixtures = @team.css('.accordion-container.upcoming')
    puts "UPCOMING FIXTURES âš½ï¸Ž | #{@font_lg} color=#{@font_color}" if fixtures
    # puts "| templateImage=\"#{icon}\" "
    fixtures.css("span[typeof=SportsEvent]").each do |item|
      date = item.css(".date span[property=startDate]").first.text
      league = item.css(".date span[property=superEvent]").first.text
      fixture = item.css("ul.upcoming-fixtures").first
      datetime = "#{date}T#{fixture.css(".status.upcoming abbr").first.text}"
      puts "#{fixture.text.strip.gsub(/(\s*\d{2}:\d{2})/, '').upcase}  | #{@font_md} color=#{@font_color}" if fixture
      puts "#{fixture_time(datetime).upcase} â€£ #{fixture_date(datetime).upcase} â€£ #{league.upcase} | #{@font_rg}" if date && league

    end
    puts "---" if fixtures
  end

  def previous_fixtures
    fixtures = @team.css('.accordion-container.previous')
    puts "RESULTS âš½ï¸Ž | #{@font_lg} color=#{@font_color}" if fixtures
    fixtures.css("span[typeof=SportsEvent]").each do |item|
      date = item.css(".date span[property=startDate]").first.text
      league = item.css(".date span[property=superEvent]").first.text
      fixture = item.css("ul.previous-fixtures").first
      datetime = "#{date}"
      link = fixture.css("a").first['href']

      puts "#{fixture.text.strip.gsub(/(\s*FT)/, '')}" if fixture
      puts "--MATCH INFO âš½ï¸Ž | #{@font_md} color=#{@font_color}"
      puts "--#{fixture_date(datetime)}"
      puts "--#{league}"
      puts "--BBC Sport Recap ðŸ“° | href=http://www.bbc.com#{link}"
    end
    puts "---" if fixtures
  end

  def headlines
    lead = @team.css('#lead-item h2')
    headlines = @team.css('#more-headlines ul')
    puts "HEADLINES âš½ï¸Ž | #{@font_lg} color=#{@font_color}" if headlines
    puts "#{lead.css('a').first.text}| href=http://www.bbc.com#{lead.css('a').first['href']}"
    headlines.css("li").take(5).each do |item|
      title = item.css("a").first.text
      link = item.css("a").first['href']
      puts "#{title} | href=http://www.bbc.com#{link}"
    end
    puts "---" if headlines
  end

  def convert_datetime(datetime)
    timezone_offset = Time.now.in_time_zone('Europe/London').strftime("%z")
    DateTime.parse("#{datetime}#{timezone_offset}").to_time.in_time_zone
  end

  def fixture_time(datetime)
    @is_usa ? convert_datetime(datetime).strftime("%l:%M%P") : convert_datetime(datetime).strftime("%H:%M")
  end

  def fixture_date(datetime)
    convert_datetime(datetime).strftime("%a %b %d, %Y")
  end

  def matchday
    @team.css('.accordion-container.live-today')
  end
