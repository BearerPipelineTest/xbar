function isNew(xml) {
  const updatedOn = xml.match(/<updated>(.*?)Z<\/updated>.*/);
  if (!updatedOn || !updatedOn[1]) return false;

  const updatedDate = new Date(updatedOn[1]);
  return updatedDate.getDay() === new Date().getDay();
}
