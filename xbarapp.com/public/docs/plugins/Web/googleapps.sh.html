function openBrowser() {
  if [[ -d /Applications/Chromium.app ]]; then
    BROWSER="/Applications/Chromium.app"
  elif [[ -d /Applications/Google\ Chrome.app ]]; then
    BROWSER="/Applications/Google Chrome.app"
  fi
  open -n -a "${BROWSER}" --args --app="${1}"
}
